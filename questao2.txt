'''
Versão Recursiva
'''

def min_moedas(qtd, lista):
    if qtd < 0:
        return None
    if qtd == 0:
        return 0
    tentativas = (min_moedas(qtd - item, lista) for item in lista)
    try:
        return 1 + min(ten for ten in tentativas if ten is not None)
    except ValueError:
        return None   


print("A menor quantidade é de :", min_moedas(20, [1,3,5,10]))
